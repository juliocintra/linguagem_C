/* Ler uma matriz 3x3 de inteiros e, após a leitura, 
zerar os valores negativos da matriz e mostrá-la no final.

Mat = 

1  0  6

4  5  7

0  7  9

*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int mat[3][3], i, j;
	
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
		{
			printf("Mat[%d][%d]: ", i, j);
			scanf("%d", &mat[i][j]);
		}
		printf("\n");
	}

	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
		{
			if (mat[i][j] < 0)
				mat[i][j] = 0;
		}
	}

	printf("\n\nMat =\n\n");
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
		{
			printf("%d  ", mat[i][j]);
		}
		printf("\n\n");	
	}
	
	printf("\n\n\n");
	system("pause");
}
