/* Ler uma matriz 3x4 de inteiros, calcular e mostrar, após
a leitura, a média aritmética dos valores dessa matriz.
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int mat[3][4], i, j;
	float soma, media;
	
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 4; j++)
		{
			printf("Mat[%d][%d]: ", i, j);
			scanf("%d", &mat[i][j]);
		}
		printf("\n");
	}
	
	soma = 0;
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 4; j++)
		{
			soma = soma + mat[i][j];
		}
	}
	
	media = soma / 12;
	
	printf("\n\nMedia aritmetica da matriz = %.3f", media);

	printf("\n\n\n");
	system("pause");	
}
