/* Ler uma matriz 4x4 de inteiros e, após a leitura, 
calcular e mostrar a qtde de pares e a qtde de ímpares 
contidos nessa matriz.
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int mat[4][4], i, j, qp, qi;
	
	for (i = 0; i < 4; i++)
	{
		for (j = 0; j < 4; j++)
		{
			printf("Mat[%d][%d]: ", i, j);
			scanf("%d", &mat[i][j]);
		}
		printf("\n");
	}

	qp = qi = 0;
	for (i = 0; i < 4; i++)
	{
		for (j = 0; j < 4; j++)
		{
			if (mat[i][j] % 2 == 0)
				qp++;
			else
				qi++;
		}	
	}
	
	printf("\n\nQtde de pares = %d", qp);
	printf("\n\nQtde de impares = %d", qi);
	
	printf("\n\n\n");
	system("pause");
	
	
}
