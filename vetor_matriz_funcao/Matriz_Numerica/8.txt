/* Ler uma matriz 5x5 de inteiros, e após a leitura, calcular
e mostrar a soma dos valores da diagonal principal e o
produto dos valores da diagonal secundária da matriz.
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int mat[5][5], i, j, soma, prod;
	
	for (i = 0; i < 5; i++)
	{
		for (j = 0; j < 5; j++)
		{
			printf("Mat[%d][%d]: ", i, j);
			scanf("%d", &mat[i][j]);
		}
		printf("\n");
	}

	soma = 0;
	for (i = 0; i < 5; i++)
		soma = soma + mat[i][i];
		
	printf("\n\nSoma da diagonal principal = %d", soma);

	prod = 1;
	for (i = 0; i < 5; i++)
		prod = prod * mat[i][4 - i];
		
	printf("\n\nProduto da diagonal secundaria = %d", prod);
	
	printf("\n\n\n");
	system("pause");
}
