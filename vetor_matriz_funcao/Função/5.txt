/* Criar uma função para zerar um vetor passado como parâmetro
juntamente com seu tamanho. O main deve chamar a função e 
mostrar o conteúdo do vetor no final.
*/

#include <stdio.h>
#include <stdlib.h>

void zerarVetor(int pvet[], int ptam)
{
	int i;
	
	for (i = 0; i < ptam; i++)
	{
		pvet[i] = 0;	
	}
}

main()
{
	int va[10], vb[7], i;
	
	zerarVetor(va, 10);
	printf("\n\nVetor A = ");
	for (i = 0; i < 10; i++)
	{
		printf("%d  ", va[i]);	
	}

	zerarVetor(vb, 7);
	printf("\n\nVetor B = ");
	for (i = 0; i < 7; i++)
	{
		printf("%d  ", vb[i]);	
	}
	
	printf("\n\n\n");
	system("pause");
}
