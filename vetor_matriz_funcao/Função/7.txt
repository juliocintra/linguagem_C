/*	Função que recebe um vetor de inteiros e seu tamanho, 
calcula e retorna a qtde de valores pares do vetor. O main deve 
ler um vetor com 7 valores, chamar a função e mostrar o 
resultado no final.
*/

#include <stdio.h>
#include <stdlib.h>

int qtdePar(int pvet[], int ptam)
{
	int i, qp;
	
	qp = 0;
	for (i = 0; i < ptam; i++)
	{
		if (pvet[i] % 2 == 0)
			qp++;
	}
	
	return qp;
}

main()
{
	int vet[7], qpar, i;
	
	for (i = 0; i < 7; i++)
	{
		printf("Vet[%d]: ", i);
		scanf("%d", &vet[i]);	
	}
	
	qpar = qtdePar(vet, 7);
	
	printf("\n\nQtde de pares = %d", qpar);
	
	printf("\n\n\n");
	system("pause");
}
