/*  Ler um vetor com 7 valores inteiros e, após a leitura, 
calcular e mostrar:
    
    a) O menor valor do vetor;
    
    b) A qtde de valores iguais ao menor.
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int vet[7], i, menor, qtde;
	
	for (i = 0; i < 7; i++)
	{
		printf("Vet[%d]: ", i);
		scanf("%d", &vet[i]);	
	}
	
	menor = vet[0];
	for (i = 1; i < 7; i++)
	{
		if (vet[i] < menor)
			menor = vet[i];	
	}
	
	printf("\n\nMenor valor do vetor = %d", menor);
	
	qtde = 0;
	for (i = 0; i < 7; i++)
	{
		if (vet[i] == menor)
			qtde++;	
	}
	
	printf("\n\nQtde de valores iguais a %d = %d", menor, qtde);
	
	printf("\n\n\n");
	system("pause");
}
