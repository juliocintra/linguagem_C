/* Ler um vetor com 10 valores inteiros e, após a leitura, 
separar os valores positivos em outro vetor. No final, 
mostrar o vetor de positivos.

  Ex.:
    
    X = 2  -4  -5  90  -1  4  6  23  -30  0
    
    P = 2  90  4  6  23  0
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int x[10], p[10], i, k;
	
	for (i = 0; i < 10; i++)
	{
		printf("X[%d]: ", i);
		scanf("%d", &x[i]);
	}
	
	k = 0;
	for (i = 0; i < 10; i++)
	{
		if (x[i] >= 0)
		{
			p[k] = x[i];
			k++;
		}
	}
	
	printf("\n\nP = ");
	for (i = 0; i < k; i++)
	{
		printf("%d  ", p[i]);	
	}
	
	printf("\n\n\n");
	system("pause");
}
