/* Ler um vetor com uma qtde qualquer de valores reais 
(float) e, após a leitura, calcular e mostrar a média 
aritmética dos valores.
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	float soma, media;
	int i, qtde;
	
	printf("Qtde de valores: ");
	scanf("%d", &qtde);
	
	float vet[qtde];
	
	system("cls");
	
	for (i = 0; i < qtde; i++)
	{
		printf("Vet[%d]: ", i);
		scanf("%f", &vet[i]);	
	}
	
	soma = 0;
	for (i = 0; i < qtde; i++)
	{
		soma = soma + vet[i];
	}
	
	media = soma / qtde;
	
	printf("\n\nMedia aritmetica do vetor = %.3f", media);
	
	printf("\n\n\n");
	system("pause");
}
