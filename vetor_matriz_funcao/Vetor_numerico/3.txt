/* Ler um vetor com 10 valores inteiros e mostrar seus
  valores separados por vírgula, sem sobrar vírgulas no
  início e no final.
  
  Ex.:
  
  Vetor = 1, 3, 6, 0, 2, 6, 9, 8, 2, 1
*/

#include <stdio.h>
#include <stdlib.h>

main()
{
	int vetor[10], i;
	
	for (i = 0; i < 10; i++)
	{
		printf("Vetor[%d]: ", i);
		scanf("%d", &vetor[i]);
	}
	
	printf("\n\nVetor = %d", vetor[0]);
	for (i = 1; i < 10; i++)
	{
		printf(", %d", vetor[i]);
	}
	
	printf("\n\n\n");
	system("pause");
}
