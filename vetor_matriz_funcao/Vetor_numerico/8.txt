/* Ler 2 vetores com 5 valores inteiros cada um, calcular a 
soma e o produto desses dois vetores e mostrar os vetores 
resultantes.

Ex.:
  
  A = 1  2  3  4  5 usuário
  
  B = 3  3  3  3  3 usuário
  
  S = 4  5  6  7  8
  
  P = 3  6  9  12  15

*/


#include <stdio.h>
#include <stdlib.h>

main()
{
	int i, a[5], b[5], s[5], p[5];
	
	for (i = 0; i < 5; i++)
	{
		printf("A[%d]: ", i);
		scanf("%d", &a[i]);
	}

	printf("\n\n");

	for (i = 0; i < 5; i++)
	{
		printf("B[%d]: ", i);
		scanf("%d", &b[i]);
	}

	for (i = 0; i < 5; i++)
	{
		s[i] = a[i] + b[i];
		p[i] = a[i] * b[i];	
	}
	
	printf("\n\nS = ");
	for (i = 0; i < 5; i++)
	{
		printf("%d  ", s[i]);
	}

	printf("\n\nP = ");
	for (i = 0; i < 5; i++)
	{
		printf("%d  ", p[i]);
	}

	printf("\n\n\n");
	system("pause");
}
